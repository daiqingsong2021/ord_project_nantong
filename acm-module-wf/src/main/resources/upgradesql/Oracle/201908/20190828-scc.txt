
DROP TABLE WSD_WF_DELEGATE;
CREATE TABLE WSD_WF_DELEGATE (
ID NUMBER(11) NOT NULL ,
ASSIGNEE NUMBER(11) NOT NULL ,
ATTORNEY NUMBER(11) NOT NULL ,
START_TIME DATE NOT NULL ,
END_TIME DATE NOT NULL ,
GLOBAL NUMBER(1) NOT NULL ,
ENABLE NUMBER(1) NULL ,
REMARKS VARCHAR2(500 BYTE) NULL ,
CREATOR NUMBER(11) NULL ,
CREAT_TIME DATE NULL ,
LAST_UPD_USER NUMBER(11) NULL ,
LAST_UPD_TIME DATE NULL ,
LAST_UPD_IP VARCHAR2(100 BYTE) NULL ,
WSDVER NUMBER(11) NULL ,
SORT_NUM NUMBER(11) NULL 
);
COMMENT ON COLUMN WSD_WF_DELEGATE.ID IS ' 主键 ';
COMMENT ON COLUMN WSD_WF_DELEGATE.ASSIGNEE IS '委托人';
COMMENT ON COLUMN WSD_WF_DELEGATE.ATTORNEY IS '代理人';
COMMENT ON COLUMN WSD_WF_DELEGATE.START_TIME IS '开始时间';
COMMENT ON COLUMN WSD_WF_DELEGATE.END_TIME IS '结束时间';
COMMENT ON COLUMN WSD_WF_DELEGATE.GLOBAL IS '是否全局';
COMMENT ON COLUMN WSD_WF_DELEGATE.ENABLE IS '启用';
COMMENT ON COLUMN WSD_WF_DELEGATE.REMARKS IS '备注';
COMMENT ON COLUMN WSD_WF_DELEGATE.CREATOR IS '创建人';
COMMENT ON COLUMN WSD_WF_DELEGATE.CREAT_TIME IS '创建时间';
COMMENT ON COLUMN WSD_WF_DELEGATE.LAST_UPD_USER IS '更新人';
COMMENT ON COLUMN WSD_WF_DELEGATE.LAST_UPD_TIME IS '更新时间';
COMMENT ON COLUMN WSD_WF_DELEGATE.LAST_UPD_IP IS 'IP';
COMMENT ON COLUMN WSD_WF_DELEGATE.WSDVER IS '版本';
COMMENT ON COLUMN WSD_WF_DELEGATE.SORT_NUM IS '排序';

ALTER TABLE WSD_WF_DELEGATE ADD CHECK (ID IS NOT NULL);
ALTER TABLE WSD_WF_DELEGATE ADD CHECK (ASSIGNEE IS NOT NULL);
ALTER TABLE WSD_WF_DELEGATE ADD CHECK (ATTORNEY IS NOT NULL);
ALTER TABLE WSD_WF_DELEGATE ADD CHECK (START_TIME IS NOT NULL);
ALTER TABLE WSD_WF_DELEGATE ADD CHECK (END_TIME IS NOT NULL);
ALTER TABLE WSD_WF_DELEGATE ADD CHECK (GLOBAL IS NOT NULL);

ALTER TABLE WSD_WF_DELEGATE ADD PRIMARY KEY (ID);

COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.DELEGATE_ID IS '代理ID';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.BIZ_TYPE_CODE IS '业务代码';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.CREATOR IS '创建人';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.CREAT_TIME IS '创建时间';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.LAST_UPD_USER IS '更新人';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.LAST_UPD_TIME IS '更新时间';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.LAST_UPD_IP IS 'IP';
COMMENT ON COLUMN WSD_WF_DELEGATE_PROC.WSDVER IS '版本';

LOGGING
VISIBLE;
